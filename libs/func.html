# 生成一个虚假的 md5 值

/**
 * 生成一个虚假的 md5 值
 */
function fake_md5(): string
{
    $chars = '0123456789abcdef';
    $fake_md5 = '';
    for ($i = 0; $i < 32; $i++) $fake_md5 += $chars[mt_rand(0, 15)];
    return $fake_md5;
}
